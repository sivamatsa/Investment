Google Sheets — Spreadsheet Backend
Use a Google Sheet as your database. Your data lives in Excel-like cells in Google Drive. Requires a one-time Apps Script setup (5 minutes). Best for Excel lovers.
Free Cross-Device 5-min setup
Open Google Sheets → create a new spreadsheet named OxyLoans Portfolio.
In the sheet, click Extensions → Apps Script. Delete all existing code.
Paste the code below into the Apps Script editor and click 💾 Save.
Click Deploy → New Deployment → Type: Web App → Execute as: Me → Access: Anyone → Deploy. Copy the Web App URL.
Paste the URL below and click Save.
function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Data') || ss.insertSheet('Data'); if (e.parameter.action === 'load') { var data = sheet.getDataRange().getValues(); if (data.length < 2) return ok([]); var headers = data[0]; var rows = data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); return ok(rows); } return ok([]); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Data') || ss.insertSheet('Data'); var payload = JSON.parse(e.postData.contents); if (payload.action === 'save' && Array.isArray(payload.data)) { sheet.clearContents(); if (payload.data.length === 0) return ok('cleared'); var keys = Object.keys(payload.data[0]); sheet.appendRow(keys); payload.data.forEach(function(row) { sheet.appendRow(keys.map(function(k) { return row[k] || ''; })); }); return ok('saved ' + payload.data.length + ' rows'); } return ok('no action'); } function ok(data) { return ContentService .createTextOutput(JSON.stringify({ success: true, data: data })) .setMimeType(ContentService.MimeType.JSON); }
